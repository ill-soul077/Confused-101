// arithmetic and power
In result @ x add 5 mul 2#
In pw @ Exp(2 $ 8)#
Show(result)#
Show(pw)#

// conditional
Iff(x grt 10) {
    Show("x is big")#
} or {
    Show("x is small")#
}

// switch
In grade @ 2#
incident(grade) {
    check 1 : { Show("fail")# }
    check 2 : { Show("pass")# }
    nop      { Show("unknown")# }
}

// range loop
flp(0:4) {
    Show(i)#
}

// while loop
In counter @ 0#
wlp(counter lrt 3) {
    counter @@ 1#
    Show(counter)#
}

// function
In square(In n) {
    return n mul n#
}
In sq @ square(5)#
Show(sq)#
